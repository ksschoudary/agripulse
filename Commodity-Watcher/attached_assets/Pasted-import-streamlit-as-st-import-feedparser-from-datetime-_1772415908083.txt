@st.cache_data(ttl=60) 
def fetch_stable_news(query, time_range="d1", limit=100):
    """Siloed Engine: Optimized for 24h-48h freshness across 28 commodities."""
    if any(x in query for x in ["Weather", "El Nino", "Monsoon"]):
        query += " -price -mandi -rate -trading -Pakistan -Europe"
    try:
        url = f"https://news.google.com/rss/search?q={query.replace(' ', '+')}&hl=en-IN&gl=IN&ceid=IN:en&tbs=qdr:{time_range}"
        feed = feedparser.parse(url)
        return sorted(feed.entries, key=lambda x: x.published_parsed if hasattr(x, 'published_parsed') else 0, reverse=True)[:limit]
    except: return []

def get_metadata(pub_date):
    """v115 Breaking News Logic: 60-minute real-time trigger."""
    diff = datetime.utcnow() - pub_date
    minutes = diff.seconds // 60
    if diff.days == 0 and minutes <= 60:
        return f"🚨 BREAKING | {minutes}m ago", "breaking-badge"
    if diff.days > 0: return f"{diff.days}d ago", "meta-badge"
    hours = diff.seconds // 3600
    if hours > 0: return f"{hours}h ago", "meta-badge"
    return f"{minutes}m ago", "meta-badge"
